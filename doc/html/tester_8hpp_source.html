95 // located
96 std::filesystem::path look(".");
97 while (look.has_relative_path()) {
98 if (std::filesystem::exists(look / "supra.toml")) {
99 break;
100 }
101 look = look / "..";
102 }
103 std::string path = "tests/" + name + ".cpp";
104
105 if (std::filesystem::exists(path) && !(std::filesystem::is_empty(path))) {
106 throw testing::SupraWriterException(
107 "Error: Test <" + path + "> already exists and is not an empty file");
108 } else {
109 // create the tests directory if it doesn't exist
110 if (!std::filesystem::exists("tests")) {
111 std::filesystem::create_directory("tests");
112 }
113 // create the test file
114 std::ofstream test_file;
115 test_file.open(path);
